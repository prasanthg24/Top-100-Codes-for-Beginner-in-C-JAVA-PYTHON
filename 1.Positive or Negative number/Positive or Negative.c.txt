#include <stdio.h>

int main() {
    int n;
    printf("Enter the number = ");
    scanf("%d",&n);
    if(n>0)
    {
        printf("%d is a Positive number",n);
        
    }
    else
    {
        if(n==0)
        {
            printf("%d is Zero",n);
        }
        else{
            printf("%d is a Negative number",n);
        }
    }

    return
}
