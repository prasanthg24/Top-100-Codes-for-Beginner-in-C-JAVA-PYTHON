import java.util.Scanner;
public class MyClass {
    public static void main(String args[])
    {
        System.out.println("Enter the number = ");
        Scanner sc =new Scanner(System.in);
        int n=sc.nextInt();
        int sum =(n*(n+1))/2;
        System.out.println("Sum of First N Natural numbers = "+sum);
      
    }
}
